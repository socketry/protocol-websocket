def initialize(finished = true, payload = nil, opcode: self.class::OPCODE, mask: false)
	if mask == true
		mask = SecureRandom.bytes(4)
	end
	
	@finished = finished
	@opcode = opcode
	@mask = mask
	@length = payload&.bytesize
	@payload = payload
end
